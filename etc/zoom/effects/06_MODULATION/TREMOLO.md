void process(float *in, float *out, int n, State *s, Params *p) {
  for (int i = 0; i < n; i++) {
    s->lfo_phase += p->rate / SAMPLE_RATE;
    float lfo = generate_lfo(s->lfo_phase, p->wave); // sin/tri/sq
    float mod = 1.0f - p->depth * (1.0f - lfo) * 0.5f;
    out[i] = in[i] * mod;
  }
}
