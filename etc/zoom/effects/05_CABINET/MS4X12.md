#### 疑似コード

```c
void process(float *in, float *out, int n, State *s, Params *p) {
  for (int i = 0; i < n; i++) {
    float x = in[i];
    // Multi-stage biquad EQ (cabinet IR approximation)
    for (int j = 0; j < NUM_BIQUADS; j++)
      x = biquad_process(&s->eq[j], x);
    out[i] = x * p->level;
  }
}
```
