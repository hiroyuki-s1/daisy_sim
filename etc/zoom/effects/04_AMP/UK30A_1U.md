#### 疑似コード

```c
// AC30 アンプモデル DSP処理
void process(float *in, float *out, int nsamples, State *s, Params *p) {
  for (int n = 0; n < nsamples; n++) {
    float x = in[n];

    // 1. Input HPF (DC blocking)
    x = biquad_process(&s->input_hpf, x);

    // 2. Gain依存 Pre-EQ
    x = parametric_eq(&s->pre_eq, x);  // 係数はGainパラメータで補間

    // 3. 4x Upsample (48→192kHz)
    float ovs_buf[4];
    upsample_4x(x, ovs_buf, &s->upsample_filter);

    for (int k = 0; k < 4; k++) {
      float y = ovs_buf[k];

      // 4. 多段ゲイン + 段間フィルタ + クリッパー
      for (int stage = 0; stage < NUM_GAIN_STAGES; stage++) {
        y = biquad_process(&s->ovs_eq[stage], y);
        y *= s->gain[stage];  // dBからリニア変換済み
        y = hard_clip(y, s->clip_threshold[stage]);
      }

      // 5. Presence EQ (OVSドメイン)
      y = parametric_eq(&s->presence_eq, y);

      ovs_buf[k] = y;
    }

    // 6. 4x Downsample (192→48kHz)
    x = downsample_4x(ovs_buf, &s->downsample_filter);

    // 7. Tone Stack (3バンド)
    x = tonestack_process(&s->tonestack, x, p->bass, p->mid, p->treble);

    // 8. Output Level
    out[n] = x * p->output_level;
  }
}

// ハードクリッパー (対称)
float hard_clip(float x, float threshold) {
  if (x > threshold) return threshold;
  if (x < -threshold) return -threshold;
  return x;
}
```
